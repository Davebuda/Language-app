<!-- Types all exercise components share -->
import type { SessionItem, ExerciseResult } from '@/types/session';
import type { Sentence } from '@/types/content';

<!-- ExerciseResult shape — onResult callback produces this -->
interface ExerciseResult {
  sessionId: string;
  itemId: string;
  correct: boolean;
  userAnswer: string;
  correctAnswer: string;
  timeTakenSeconds: number;
  errorTag?: ErrorTag;   // set if wrong
  conceptId: string;
}

<!-- All exercise components share this prop shape -->
interface ExerciseProps {
  item: SessionItem;       // contains exerciseType, contentId, conceptIds
  sentence: Sentence;      // the actual content (norwegian, english, audioUrl, etc.)
  sessionId: string;
  onResult: (result: ExerciseResult) => void;
}

<!-- ExerciseCard receives these props from SessionScreen -->
interface ExerciseCardProps {
  item: SessionItem;
  sentence: Sentence;
  sessionId: string;
  onResult: (result: ExerciseResult) => void;
  repairPlan?: RepairPlan | null;    // set when isInRepair; triggers ExplanationCard
}

<!-- ProgressBar props -->
interface ProgressBarProps {
  current: number;   // 0-based index of current item
  total: number;     // total items in session
}

<!-- Answer normalization (shared utility — create in src/lib/answer.ts) -->
function normalizeAnswer(s: string): string {
  return s.trim().toLowerCase().replace(/[.,!?;:]/g, '');
}
function checkAnswer(user: string, correct: string): boolean {
  return normalizeAnswer(user) === normalizeAnswer(correct);
}
</interfaces>
