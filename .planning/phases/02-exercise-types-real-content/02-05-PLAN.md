      ```typescript
      type MasteryState = 'locked' | 'learning' | 'practiced' | 'mastered';

      function getMasteryState(
        conceptId: string,
        mastery: ConceptMastery | undefined,
        masteryThreshold: number,
        prerequisites: string[],
        masteredIds: Set<string>
      ): MasteryState {
        const prereqsMet = prerequisites.every(p => masteredIds.has(p));
        if (!mastery && !prereqsMet) return 'locked';
        if (!mastery) return 'learning';   // unlocked but not started = ready to learn
        if (isMastered(mastery, masteryThreshold, 20, 4)) return 'mastered';
        if (mastery.decayedScore >= 50) return 'practiced';
        return 'learning';
      }
      ```
